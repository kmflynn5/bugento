set guifont=Inconsolata\ for\ Powerline:h15

   set background=dark
   colorscheme solarized
   let s:uname = system("uname")

      set guifont=Inconsolata\ for\ Powerline:h15
   endif

" Enable folding with the spacebar

" PEP 8 Python formatting settings
au BufNewFile,BufRead *.py
    \ set tabstop=4 |
    \ set softtabstop=4 |
    \ set shiftwidth=4 |
    \ set textwidth=79 |
    \ set expandtab |
    \ set autoindent |
    \ set fileformat=unix

    \ set softtabstop=2 |
    \ set shiftwidth=2
